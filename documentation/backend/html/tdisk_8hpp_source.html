600  tDiskPerformanceImprovement getPerformanceImprovement(std::size_t amountFiles)
601  {
602  unsigned int fastestDevice = 1;
603  double bestPerformance = 999999999;
604  long double avgPerformance = 0;
605  const unsigned int devices = getInternalDevicesCount();
606  for(unsigned int device = 1; device <= devices; ++device)
607  {
608  f_internal_device_info info = getDeviceInfo(device);
609 
610  double avg_read_dec = (double)info.performance.mod_avg_read / (1 << c::get_measure_records_shift());
611  double avg_write_dec = (double)info.performance.mod_avg_write / (1 << c::get_measure_records_shift());
612 
613  double avg_read_time_cycles = (double)info.performance.avg_read_time_cycles + avg_read_dec;
614  double avg_write_time_cycles = (double)info.performance.avg_write_time_cycles + avg_write_dec;
615 
616  double performance = avg_read_time_cycles + avg_write_time_cycles;
617  if(performance < bestPerformance)
618  {
619  bestPerformance = performance;
620  fastestDevice = device;
621  }
622 
623  avgPerformance += performance;
624  }
625  avgPerformance /= devices;
626 
627  std::vector<FileAssignment> files = getFilesOnDevice(fastestDevice, true, true);
628 
629  long double avgPercentage = 0;
630  for(const FileAssignment &f : files)
631  {
632  avgPercentage += f.percentage;
633  }
634  avgPercentage /= files.size();
635 
636  long double performanceDifference = (avgPerformance - bestPerformance) / avgPerformance;
637 
638  tDiskPerformanceImprovement p((double)(performanceDifference * avgPercentage));
639  for(std::size_t i = 0; i < amountFiles && i < files.size(); ++i)
640  {
641  const FileAssignment &f = files[i];
642  p.addFile(f.filename, (double)(f.percentage * performanceDifference));
