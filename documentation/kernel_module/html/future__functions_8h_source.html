29 /*
30  * loop_switch performs the hard work of switching a backing store.
31  * First it needs to flush existing IO, it does this by sending a magic
32  * BIO down the pipe. The completion of this BIO does the actual switch.
33  */
34 static int loop_switch(struct loop_device *lo, struct file *file)
35 {

37 
38  w.file = file;

40  //freeze queue and wait for completion of scheduled requests

42 
43  //do the switch action
44  do_loop_switch(lo, &w);
45 
46  //unfreeze
47  blk_mq_unfreeze_queue(lo->lo_queue);
48 
49  return 0;
50 }

60 /*
61  * loop_change_fd switched the backing store of a loopback device to
62  * a new file. This is useful for operating system installers to free up
63  * the original file and in High Availability environments to switch to
64  * an alternative location for the content in case of server meltdown.
65  * This can only work if the loop device is used read-only, and if the
66  * new backing store is the same size and type as the old backing store.
67  */
68 static int loop_change_fd(struct loop_device *lo, struct block_device *bdev,
69  unsigned int arg)
70 {
71  struct file *file, *old_file;
72  struct inode *inode;
73  int error;
74 
75  error = -ENXIO;
76  if(lo->lo_state != Lo_bound)
77  goto out;
78 
79  //the loop device has to be read-only
80  error = -EINVAL;
81  if(!(lo->lo_flags & LO_FLAGS_READ_ONLY))
82  goto out;
83 
84  error = -EBADF;
85  file = fget(arg);
86  if(!file)
87  goto out;
88 
89  inode = file->f_mapping->host;
90  old_file = lo->lo_backing_file;
91 
92  error = -EINVAL;
93 
94  if(!S_ISREG(inode->i_mode) && !S_ISBLK(inode->i_mode))
95  goto out_putf;
96 
97  //size of the new backing store needs to be the same
98  if(get_loop_size(lo, file) != get_loop_size(lo, old_file))
99  goto out_putf;
100 
101  //and ... switch
102  error = loop_switch(lo, file);
103  if(error)
104  goto out_putf;
105 
106  fput(old_file);
107  if(lo->lo_flags & LO_FLAGS_PARTSCAN)
108  ioctl_by_bdev(bdev, BLKRRPART, 0);
109 
110  return 0;
111 
112  out_putf:
113  fput(file);
114  out:
115  return error;
116 }
